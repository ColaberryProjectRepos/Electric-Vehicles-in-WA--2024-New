 Total Vehicles  Measure
Total Vehicles = COUNT (Electric_Vehicle_Data[DOL Vehicle ID])
Average Electric Range

CAFV Eligibility Measure
CAFV_Eligibility = 
VAR Total_EVs = COUNTROWS(FILTER('Electric_Vehicle_Data', NOT(ISBLANK('Electric_Vehicle_Data'[DOL Vehicle ID]))))
VAR Eligible_EVs = COUNTROWS(FILTER('Electric_Vehicle_Data', 'Electric_Vehicle_Data'[Clean Alternative Fuel Vehicle (CAFV) Eligibility] = "Clean Alternative Fuel Vehicle Eligible"))
RETURN 
IF(Total_EVs = 0, BLANK(), (Eligible_EVs / Total_EVs) )
BEV COUNT(To be used in creating BEV Dist.)
BEVCount = 
CALCULATE(
    COUNTROWS(Electric_Vehicle_Data),
    Electric_Vehicle_Data[Electric Vehicle Type] = "Battery Electric Vehicle (BEV)"
)
BEV Distribution
BEV Distribution = DIVIDE(Electric_Vehicle_Data[BEVCount],Electric_Vehicle_Data[Total Vehicles])
